    function _rdEditDone(result,ctx) {
      var btn=document.getElementById('rd_saveEditBtn'); btn.disabled=false; btn.textContent='💾 Save Changes';
      if (!result||!result.ok){toast('Save failed','error');return;}
      toast('Changes saved!','success');
      var loadMap={meal:'getMealsSection',bh:'getBhSection',quality:'getQualitySection',ingredient:'getIngredientsSection',fix:'getFixesSection'};
      var loadFn=loadMap[ctx.type];
      if (!loadFn){closeRowDetailsModal();return;}
      google.script.run.withSuccessHandler(function(data){
        if(data.dashboard){DASH=data.dashboard;} if(data.performers){PERFORMERS=data.performers;}
        if(data.bh){BH=data.bh;} if(data.quality){QUALITY=data.quality;}
        if(data.ingredients){ING=data.ingredients;} if(data.fixes){FIXES=data.fixes;}
        closeRowDetailsModal();
        var renderMap={meal:renderMeals,bh:renderBh,quality:renderQuality,ingredient:renderIngredients,fix:renderFixes};
        if(renderMap[ctx.type]) renderMap[ctx.type]();
      })[loadFn]();
    }

    function _findRowItem(type, row, sheetName) {
      try {
        if (type === 'meal') {
          var meals = (DASH && DASH.meals) || [];
          for (var i = 0; i < meals.length; i++) if (meals[i].row === row) return meals[i];
        } else if (type === 'bh') {
          var bm = (BH && BH.meals) || [];
          for (var j = 0; j < bm.length; j++) if (bm[j].row === row) return bm[j];
        } else if (type === 'ingredient') {
          var items = (ING && ING.items) || [];
          for (var k = 0; k < items.length; k++) if (items[k].row === row) return items[k];
        } else if (type === 'fix') {
          var fitems = (FIXES && FIXES.items) || [];
          for (var l = 0; l < fitems.length; l++) if (fitems[l].row === row) return fitems[l];
        } else if (type === 'quality') {
          var qs = (QUALITY && QUALITY.entries) || [];
          for (var m = 0; m < qs.length; m++) {
            if (qs[m].row === row && (!sheetName || qs[m].sheetName === sheetName)) return qs[m];
          }
        }
      } catch(e) {}
      return null;
    }

    function submitRowDetailsPhoto() {
      if (!RD_CONTEXT) return;
      var file = (document.getElementById('rd_photoFile').files || [])[0];
      if (!file) { toast('Please select a photo first', 'error'); return; }
      if (file.size > 10 * 1024 * 1024) { toast('Photo too large (max 10MB)', 'error'); return; }
      
      // Overwrite confirmation
      var hasExisting = false;
      var slotChoice = '';
      if (RD_CONTEXT.type === 'quality') {
        slotChoice = document.getElementById('rd_qualitySlot').value;
        var d = RD_CONTEXT.dataItem;
        if      (slotChoice === 'photo1' && d.photo1Url)    hasExisting = true;
        else if (slotChoice === 'photo2' && d.photo2Url)    hasExisting = true;
        else if (slotChoice === 'final'  && d.finalPhotoUrl) hasExisting = true;
      } else {
        if (RD_CONTEXT.currentPhotoUrl) hasExisting = true;
      }
      if (hasExisting && !confirm('A photo already exists for this row. Overwrite it?')) return;
      
      var saveBtn = document.getElementById('rd_saveBtn');
      var ctx = RD_CONTEXT;  // Snapshot — RD_CONTEXT may be cleared by modal close
      
      saveBtn.disabled = true; saveBtn.textContent = 'Uploading…';
      
      // Safety net: if no response in 90s, re-enable button
      var safetyTimer = setTimeout(function() {
        saveBtn.disabled = false; saveBtn.textContent = '💾 Save Photo';
        toast('Upload is taking longer than expected. Try again, or check the sheet directly.', 'error');
      }, 90000);
      
      function resetButton() {
        clearTimeout(safetyTimer);
        saveBtn.disabled = false; saveBtn.textContent = '💾 Save Photo';
      }
      
      function onSuccess(sectionKey) {
        return function(result) {
          clearTimeout(safetyTimer);
          try {
            var newUrl = result && result.photoUrl;
            if (!newUrl) {
              toast('Upload OK but no URL returned. Refreshing section…', 'success');
              closeRowDetailsModal();
              resetButton();
              refreshSection(sectionKey);
              return;
            }
            
            // Patch in-memory row data
            if (sectionKey === 'meals' && DASH && DASH.meals) {
              for (var i = 0; i < DASH.meals.length; i++) {
                if (DASH.meals[i].row === ctx.row) {
                  DASH.meals[i].photoUrl = newUrl;
                  DASH.meals[i].photoEmbedded = false;
                  break;
                }
              }
            } else if (sectionKey === 'bh' && BH && BH.meals) {
              for (var j = 0; j < BH.meals.length; j++) {
                if (BH.meals[j].row === ctx.row) {
                  BH.meals[j].photoUrl = newUrl;
                  BH.meals[j].photoEmbedded = false;
                  break;
                }
              }
            } else if (sectionKey === 'ingredients' && ING && ING.items) {
              for (var k = 0; k < ING.items.length; k++) {
                if (ING.items[k].row === ctx.row) {
                  ING.items[k].photoUrl = newUrl;
                  ING.items[k].photoEmbedded = false;
                  break;
                }
              }
            } else if (sectionKey === 'fixes' && FIXES && FIXES.items) {
              for (var l = 0; l < FIXES.items.length; l++) {
                if (FIXES.items[l].row === ctx.row) {
                  FIXES.items[l].photoUrl = newUrl;
                  FIXES.items[l].photoEmbedded = false;
                  break;
                }
              }
            } else if (sectionKey === 'quality' && QUALITY && QUALITY.entries) {
              for (var m = 0; m < QUALITY.entries.length; m++) {
                var q = QUALITY.entries[m];
                if (q.row === ctx.row && (q.sheetName === ctx.sheetName || q.sheetName === ctx.dataItem.sheetName)) {
                  if (slotChoice === 'photo1') { q.photo1Url = newUrl; q.photo1Embedded = false; }
                  else if (slotChoice === 'photo2') { q.photo2Url = newUrl; q.photo2Embedded = false; }
                  else if (slotChoice === 'final')  { q.finalPhotoUrl = newUrl; q.finalPhotoEmbedded = false; }
                  break;
                }
              }
            }
            
            // Sync the MP_CACHE pointer (the cached object IS the same reference, but be safe)
            try {
              var cache = MP_CACHE[CURRENT_MP];
              if (cache) {
                if (sectionKey === 'meals' && cache.meals && cache.meals.dashboard) cache.meals.dashboard = DASH;
                else if (sectionKey === 'bh' && cache.bh) cache.bh.bh = BH;
                else if (sectionKey === 'quality' && cache.quality) cache.quality.quality = QUALITY;
                else if (sectionKey === 'ingredients' && cache.ingredients) cache.ingredients.ingredients = ING;
                else if (sectionKey === 'fixes' && cache.fixes) cache.fixes.fixes = FIXES;
              }
            } catch(e) { console.warn('cache sync failed', e); }
            
            // Re-render to show the new photo
            try {
              if (sectionKey === 'meals')       applyFilters();
              else if (sectionKey === 'bh')     applyBhFilters();
              else if (sectionKey === 'quality') applyQualityFilters();
              else if (sectionKey === 'ingredients') applyIngFilters();
              else if (sectionKey === 'fixes')  applyFixesFilters();
            } catch(e) { console.warn('re-render failed', e); }
            
            toast('Photo saved ✓', 'success');
            closeRowDetailsModal();
            resetButton();
          } catch(e) {
            // CRITICAL: never leave button stuck. If anything above throws, recover gracefully.
            console.error('Post-upload handler threw:', e);
            resetButton();
            closeRowDetailsModal();
            toast('Saved, but UI sync failed. Refresh the page to see the photo.', 'error');
          }
        };
      }
      
      function onFailure(err) {
        clearTimeout(safetyTimer);
        saveBtn.disabled = false; saveBtn.textContent = '💾 Save Photo';
        var msg = err && err.message ? err.message : String(err);
        console.error('updatePhoto failed:', msg);
        toast('Error: ' + msg, 'error');
      }
      
      var reader = new FileReader();
      reader.onload = function(e) {
        var photoData = { bytes: e.target.result.split(',')[1], mimeType: file.type, name: file.name };
        
        if (ctx.type === 'meal') {
          google.script.run.withSuccessHandler(onSuccess('meals')).withFailureHandler(onFailure).updateMealPhoto(ctx.row, photoData);
        } else if (ctx.type === 'bh') {
          google.script.run.withSuccessHandler(onSuccess('bh')).withFailureHandler(onFailure).updateBhPhoto(ctx.row, photoData);
        } else if (ctx.type === 'ingredient') {
          google.script.run.withSuccessHandler(onSuccess('ingredients')).withFailureHandler(onFailure).updateIngPhoto(ctx.row, photoData);
        } else if (ctx.type === 'fix') {
          google.script.run.withSuccessHandler(onSuccess('fixes')).withFailureHandler(onFailure).updateFixPhoto(ctx.row, photoData);
        } else if (ctx.type === 'quality') {
          google.script.run.withSuccessHandler(onSuccess('quality')).withFailureHandler(onFailure)
            .updateQualityPhoto(ctx.sheetName || ctx.dataItem.sheetName, ctx.row, slotChoice, photoData);
        } else {
          resetButton();
          toast('Unknown row type', 'error');
        }
      };
      reader.onerror = function() {
        clearTimeout(safetyTimer);
        saveBtn.disabled = false; saveBtn.textContent = '💾 Save Photo';
        toast('Failed to read photo file', 'error');
      };
      reader.readAsDataURL(file);
    }

    /* ===== Lazy section loaders ===== */
    function ensureSectionLoaded(section, forceReload) {
      var cache = _ensureMpCache_(CURRENT_MP);
      var cachedKey = section;
      
      if (!forceReload && cache[cachedKey]) {
        applySectionData(section, cache[cachedKey]);
        return;
      }
      if (LOADING_SECTION[CURRENT_MP + ':' + section]) return;  // Already fetching
      LOADING_SECTION[CURRENT_MP + ':' + section] = true;
      
      var sectionEl = document.getElementById('view_' + section);
      _setLoading_(sectionEl, true, SECTION_LABELS[section] || section);
      
      var rpcName = {
        meals: 'getMealsSection',
        bh: 'getBhSection',
        quality: 'getQualitySection',
        ingredients: 'getIngredientsSection',
        fixes: 'getFixesSection',
        audits: 'getAuditsSection'
      }[section];
      
      google.script.run
        .withSuccessHandler(function(res) {
          delete LOADING_SECTION[CURRENT_MP + ':' + section];
          _setLoading_(sectionEl, false);
          cache[cachedKey] = res;
          applySectionData(section, res);
        })
        .withFailureHandler(function(err) {
          delete LOADING_SECTION[CURRENT_MP + ':' + section];
          _setLoading_(sectionEl, false);
          toast('Failed to load ' + section + ': ' + (err && err.message ? err.message : String(err)), 'error');
        })
        [rpcName]();
    }
    
    function applySectionData(section, data) {
      try {
        if (section === 'meals') {
          DASH = data.dashboard || { meals:[], totals:{}, filters:{} };
          PERFORMERS = data.performers || null;
          renderKPIs(DASH.totals); renderPerformers(PERFORMERS); renderCharts(DASH);
          populateFilters(DASH.filters); applyFilters();
        } else if (section === 'bh') {
          BH = data.bh || { meals:[], totals:{}, filters:{} };
          renderBhKPIs(BH.totals); renderBhCharts(BH); populateBhFilters(BH.filters); applyBhFilters();
        } else if (section === 'quality') {
          QUALITY = data.quality || { entries:[], totals:{}, filters:{}, months:[] };
          renderQualityKPIs(QUALITY.totals); renderQualityCharts(QUALITY); populateQualityFilters(QUALITY.filters);
          populateQualityForm(QUALITY.months); applyQualityFilters();
        } else if (section === 'ingredients') {
          ING = data.ingredients || { items:[], totals:{}, filters:{} };
          renderIngKPIs(ING.totals); renderIngCharts(ING); populateIngFilters(ING.filters); applyIngFilters();
        } else if (section === 'fixes') {
          FIXES = data.fixes || { items:[], totals:{}, filters:{} };
          renderFixesKPIs(FIXES.totals); renderFixesCharts(FIXES); populateFixesFilters(FIXES.filters);
          populateFixesForm(FIXES.filters); applyFixesFilters();
        } else if (section === 'audits') {
          AUDITS = data.audits || { rows:[], sessions:[], totals:{}, filters:{} };
          renderAuditsKPIs(AUDITS.totals); renderAuditsCharts(AUDITS); populateAuditsFilters(AUDITS.filters); applyAuditFilters();
        }
      } catch(e) {
        console.error('applySectionData', section, e);
      }
    }

      var sel = document.getElementById('mpSelect');
      var newMp = sel.value;
      if (!newMp || newMp === CURRENT_MP) return;
      
      sel.disabled = true;
      _showTopProgress_();
      
      google.script.run
        .withSuccessHandler(function(res) {
          _hideTopProgress_();
          CURRENT_MP = res.currentMp || newMp;
          // Always re-enable the dropdown immediately — never leave it stuck
          sel.disabled = false;
          CURRENT_MP_NAME = (function(){
            for (var i=0; i<MP_REGISTRY.length; i++) if (MP_REGISTRY[i].key === CURRENT_MP) return MP_REGISTRY[i].display;
            return CURRENT_MP;
          })();
          toast('Switched to ' + CURRENT_MP_NAME, 'success');
          _updateSplashTag_(CURRENT_MP_NAME);
          var cache = MP_CACHE[CURRENT_MP];
          if (cache && cache.meals) {
            // Cached — render instantly from memory
            applySectionData('meals', cache.meals);
            if (cache.bh) applySectionData('bh', cache.bh); else BH = null;
            if (cache.quality) applySectionData('quality', cache.quality); else QUALITY = null;
            if (cache.ingredients) applySectionData('ingredients', cache.ingredients); else ING = null;
            if (cache.fixes) applySectionData('fixes', cache.fixes); else FIXES = null;
            if (cache.audits) applySectionData('audits', cache.audits); else AUDITS = null;
            if (!cache[CURRENT_VIEW === 'ingredients' ? 'ingredients' : CURRENT_VIEW]) {
              ensureSectionLoaded(CURRENT_VIEW);
            }
          } else {
            // No cache — load fresh data for this MP
            refreshDashboard();
          }
        })
        .withFailureHandler(function(err) {
          _hideTopProgress_();
          sel.disabled = false;
          sel.value = CURRENT_MP;
          toast('Switch failed: ' + (err && err.message ? err.message : String(err)), 'error');
        })
        .setActiveMp(newMp);
    }

    function parseFoodCost(raw) {
      if (raw === null || raw === undefined || raw === '') return { display: '', pctValue: null, isHigh: false };
      var pctValue = null, displayStr = '';
      if (typeof raw === 'number') {
        if (raw > 0 && raw <= 1) pctValue = raw * 100; else pctValue = raw;
        displayStr = pctValue.toFixed(2) + '%';
      } else {
        var str = String(raw).trim();
        var cleaned = str.replace(/%/g, '').replace(/,/g, '.').trim();
        var num = parseFloat(cleaned);
        if (!isNaN(num)) {
          if (str.indexOf('%') >= 0 || num > 1) pctValue = num;
          else if (num > 0 && num <= 1) pctValue = num * 100;
          else pctValue = num;
          displayStr = pctValue.toFixed(2) + '%';
        } else { displayStr = str; pctValue = null; }
      }
      return { display: displayStr, pctValue: pctValue, isHigh: pctValue !== null && pctValue > 27 };
    }

    function applyAuditFilters() {
      if (!AUDITS || !AUDITS.rows) return;
      var q = (document.getElementById('auSearchInput').value || '').toLowerCase().trim();
      var sec = document.getElementById('auFilterSection').value;
      var aud = document.getElementById('auFilterAuditor').value;
      var res = document.getElementById('auFilterResult').value;
      var aid = document.getElementById('auFilterAuditId').value;
      VISIBLE_AU = AUDITS.rows.filter(function(r) {
        if (sec && r.section !== sec) return false;
        if (aud && r.auditor !== aud) return false;
        if (res && r.result !== res) return false;
        if (aid && r.auditId !== aid) return false;
        if (q) { var hay = (r.auditId+' '+r.auditor+' '+r.section+' '+r.question+' '+r.comment).toLowerCase();
          if (hay.indexOf(q) === -1) return false; }
        return true;
      });
      if (AU_VIEW_MODE === 'sessions') renderAuditSessions(); else renderAuditRows(VISIBLE_AU);
      document.getElementById('auResultsCount').textContent = VISIBLE_AU.length + ' of ' + AUDITS.rows.length + ' questions';
    }

    function renderAuditSessions() {
      var c = document.getElementById('auSessionsContainer');
      var sessionMap = {};
      VISIBLE_AU.forEach(function(r) {
        if (!r.auditId) return;
        if (!sessionMap[r.auditId]) {
          sessionMap[r.auditId] = { auditId: r.auditId, date: r.date, auditor: r.auditor, section: r.section,
            pdfLink: r.pdfLink, questionCount: 0, passCount: 0, failCount: 0 };
        }
        sessionMap[r.auditId].questionCount++;
        if ((r.result||'').toLowerCase() === 'yes') sessionMap[r.auditId].passCount++;
        else if ((r.result||'').toLowerCase() === 'no') sessionMap[r.auditId].failCount++;
      });
      var sessions = Object.values(sessionMap);
      sessions.sort(function(a,b){ return (b.date||'').localeCompare(a.date||''); });
      if (sessions.length === 0) {
        c.innerHTML = '<div class="empty" style="grid-column:1/-1">No audit sessions match.</div>'; return;
      }
      c.innerHTML = sessions.map(function(s) {
        var passPct = s.questionCount > 0 ? Math.round((s.passCount / s.questionCount) * 100) : 0;
        var passColor = passPct >= 80 ? '#10b981' : (passPct >= 60 ? '#f59e0b' : '#ef4444');
        var pdfBlock = '';
        if (s.pdfLink) {
          var downloadUrl = s.pdfLink.replace('/view', '/uc?export=download');
          pdfBlock = '<div class="audit-pdf-row">' +
            '<a class="audit-pdf-link" href="' + escapeHTML(s.pdfLink) + '" target="_blank">📄 View PDF</a>' +
            '<a class="audit-pdf-download" href="' + escapeHTML(downloadUrl) + '" title="Download PDF">⬇</a></div>';
        } else { pdfBlock = '<div style="font-size:11px;color:#9ca3af">No PDF</div>'; }
        return '<div class="audit-session-card">' +
          '<div class="audit-session-header"><span class="audit-session-id">' + escapeHTML(s.auditId) + '</span>' +
            '<span style="font-weight:700;color:' + passColor + '">' + passPct + '%</span></div>' +
          '<div class="audit-session-meta">' +
            (s.date ? '<span>📅 ' + escapeHTML(s.date) + '</span>' : '') +
            (s.auditor ? '<span>👤 ' + escapeHTML(s.auditor) + '</span>' : '') +
            (s.section ? '<span>📍 ' + escapeHTML(s.section) + '</span>' : '') + '</div>' +
          '<div style="font-size:13px;color:#374151"><strong>' + s.questionCount + '</strong> questions · ' +
          '<span style="color:#10b981">' + s.passCount + ' pass</span> · ' +
          '<span style="color:#ef4444">' + s.failCount + ' fail</span></div>' +
          pdfBlock + '</div>';
      }).join('');
    }

    /**
     * Compress an image File to JPEG, max 1200px wide, ~85% quality.
     * Returns a Promise-like (callback) with { bytes, mimeType, name } payload.
     * Non-images (or compression failure) pass through as-is.
     */
    function _compressImage_(file, callback) {
      if (!file) { callback(null); return; }
      var mime = (file.type || '').toLowerCase();
      // Skip compression for non-image or already-small (<300KB) files
      if (mime.indexOf('image/') !== 0 || file.size < 300 * 1024) {
        _passThrough_(file, callback);
        return;
      }
      try {
        var img = new Image();
        var url = URL.createObjectURL(file);
        img.onload = function() {
          try {
            var MAX_W = 1200;
            var scale = img.width > MAX_W ? (MAX_W / img.width) : 1;
            var w = Math.round(img.width * scale);
            var h = Math.round(img.height * scale);
            var canvas = document.createElement('canvas');
            canvas.width = w; canvas.height = h;
            var ctx = canvas.getContext('2d');
            ctx.drawImage(img, 0, 0, w, h);
            var dataUrl = canvas.toDataURL('image/jpeg', 0.85);
            URL.revokeObjectURL(url);
            callback({
              bytes: dataUrl.split(',')[1],
              mimeType: 'image/jpeg',
              name: (file.name || 'photo').replace(/\.\w+$/, '') + '.jpg'
            });
          } catch(err) {
            URL.revokeObjectURL(url);
            _passThrough_(file, callback);
          }
        };
        img.onerror = function() {
          URL.revokeObjectURL(url);
          _passThrough_(file, callback);
        };
        img.src = url;
      } catch(e) {
        _passThrough_(file, callback);
      }
    }
    
    function _passThrough_(file, callback) {
      var reader = new FileReader();
      reader.onload = function(e) {
        callback({ bytes: e.target.result.split(',')[1], mimeType: file.type || 'application/octet-stream', name: file.name });
      };
      reader.onerror = function() { callback(null); };
      reader.readAsDataURL(file);
    }

    /**
     * Wrap a Save button with a 90s safety timer. Returns a {fail, reset} pair.
     * If the response never comes back, button auto-recovers and shows an error.
     */
    function _saveBtnSafety_(saveBtn, originalText) {
      var timer = setTimeout(function() {
        saveBtn.disabled = false; saveBtn.textContent = originalText;
        toast('Upload timed out. Try again or check the sheet directly.', 'error');
      }, 90000);
      return {
        reset: function() {
          clearTimeout(timer);
          saveBtn.disabled = false; saveBtn.textContent = originalText;
        },
        fail: function(msg) {
          clearTimeout(timer);
          saveBtn.disabled = false; saveBtn.textContent = originalText;
          toast('Error: ' + (msg && msg.message ? msg.message : String(msg)), 'error');
        }
      };
    }
    function refreshDashboard() {
      // Refresh ONLY the current section. Other tabs stay cached for instant switching.
      var section = CURRENT_VIEW || 'meals';
      var cache = MP_CACHE[CURRENT_MP];
      if (cache) cache[section === 'ingredients' ? 'ingredients' : section] = null;
      
      _showTopProgress_();
      
      var currentEl = document.getElementById('view_' + section);
      _setLoading_(currentEl, true, SECTION_LABELS[section] || section);
      
      // For Meals, we need to refresh via getInitialData (which also returns meta/performers).
      // For other sections, ensureSectionLoaded(forceReload=true) does the job.
      if (section === 'meals') {
        google.script.run
          .withSuccessHandler(function(data) {
            _hideTopProgress_();
            // Patch meals + meta from initial data response without nuking other cached sections
            try {
              if (data && data.dashboard) {
                DASH = data.dashboard;
                PERFORMERS = data.performers || null;
                META = data.meta || META;
                if (cache) cache.meals = { dashboard: DASH, performers: PERFORMERS };
                renderKPIs(DASH.totals); renderPerformers(PERFORMERS); renderCharts(DASH);
                populateFilters(DASH.filters); applyFilters();
              }
            } catch(e) { console.warn('refresh meals apply:', e); }
            _setLoading_(currentEl, false);
            toast('Meals refreshed', 'success');
          })
          .withFailureHandler(function(err) {
            _hideTopProgress_();
            _setLoading_(currentEl, false);
            toast('Refresh failed: ' + (err && err.message ? err.message : String(err)), 'error');
          })
          .getInitialData();
      } else {
        // Other sections: force-reload via existing lazy loader
        ensureSectionLoaded(section, true);
        // Hide top progress when section loader completes (it manages its own spinner)
        var tick = setInterval(function() {
          if (!LOADING_SECTION[CURRENT_MP + ':' + section]) {
            clearInterval(tick);
            _hideTopProgress_();
            toast((SECTION_LABELS[section] || section) + ' refreshed', 'success');
          }
        }, 200);
        // Safety: stop ticking after 30s no matter what
        setTimeout(function() { clearInterval(tick); _hideTopProgress_(); }, 30000);
      }
    }
